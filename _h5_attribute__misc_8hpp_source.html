60  const DataSpace& mem_space = getMemSpace();
61  const details::BufferInfo<T> buffer_info(getDataType());
62 
63  if (!details::checkDimensions(mem_space, buffer_info.n_dimensions)) {
64  std::ostringstream ss;
65  ss << "Impossible to read DataSet of dimensions "
66  << mem_space.getNumberDimensions() << " into arrays of dimensions "
67  << buffer_info.n_dimensions;
68  throw DataSpaceException(ss.str());
69  }
70  details::data_converter<T> converter(mem_space);
71  read(converter.transform_read(array), buffer_info.data_type);
72  // re-arrange results
